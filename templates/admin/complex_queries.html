1. Nested Query
Query: Find patients who have appointments with busy doctors
SELECT name FROM patients
WHERE id IN (
  SELECT patient_id FROM appointments
  WHERE doctor_id IN (
    SELECT id FROM doctors WHERE availability = 'Busy'
  )
)

2. Join Query
Query: Show detailed appointment information with patient and doctor details
SELECT
  p.name as patient_name,
  d.name as doctor_name,
  d.specialization,
  a.appointment_date,
  a.status
FROM appointments a
JOIN patients p ON a.patient_id = p.id
JOIN doctors d ON a.doctor_id = d.id
ORDER BY a.appointment_date DESC

3. Aggregate Query
Query: Show revenue and appointment statistics by doctor
SELECT
  d.name,
  COUNT(a.id) as total_appointments,
  SUM(b.total_amount) as total_revenue
FROM doctors d
LEFT JOIN appointments a ON d.id = a.doctor_id
LEFT JOIN bills b ON a.id = b.appointment_id
GROUP BY d.id
